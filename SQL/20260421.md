-- 习题解答表
CREATE TABLE exercise_solutions (
id SERIAL PRIMARY KEY,
parent_id INTEGER REFERENCES exercise_solutions(id) ON DELETE CASCADE,
title VARCHAR(500) NOT NULL,
description TEXT,
solution TEXT NOT NULL,
solution_type VARCHAR(50) DEFAULT 'text',
sort_order INTEGER DEFAULT 0,
difficulty_level VARCHAR(20) DEFAULT 'medium',
category VARCHAR(100),
video_reference VARCHAR(500),
web_reference VARCHAR(500),
document_reference VARCHAR(500),
created_by VARCHAR(100),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
is_deleted BOOLEAN DEFAULT FALSE
);

-- 添加表注释
COMMENT ON TABLE exercise_solutions IS '习题解答表';

-- 添加列注释
COMMENT ON COLUMN exercise_solutions.id IS '主键ID';
COMMENT ON COLUMN exercise_solutions.parent_id IS '父习题ID，支持嵌套结构';
COMMENT ON COLUMN exercise_solutions.title IS '题目标题';
COMMENT ON COLUMN exercise_solutions.description IS '题目描述';
COMMENT ON COLUMN exercise_solutions.solution IS '习题解答';
COMMENT ON COLUMN exercise_solutions.solution_type IS '解答类型: text, markdown, html';
COMMENT ON COLUMN exercise_solutions.sort_order IS '显示顺序';
COMMENT ON COLUMN exercise_solutions.difficulty_level IS '难度级别: easy, medium, hard';
COMMENT ON COLUMN exercise_solutions.category IS '分类标签';
COMMENT ON COLUMN exercise_solutions.video_reference IS '视频资料链接';
COMMENT ON COLUMN exercise_solutions.web_reference IS '网页资料链接';
COMMENT ON COLUMN exercise_solutions.document_reference IS '文档资料链接';
COMMENT ON COLUMN exercise_solutions.created_by IS '创建者';
COMMENT ON COLUMN exercise_solutions.created_at IS '创建时间';
COMMENT ON COLUMN exercise_solutions.updated_at IS '更新时间';
COMMENT ON COLUMN exercise_solutions.is_deleted IS '软删除标记';

-- 创建索引
CREATE INDEX idx_parent_id ON exercise_solutions(parent_id);
CREATE INDEX idx_sort_order ON exercise_solutions(sort_order);
CREATE INDEX idx_difficulty ON exercise_solutions(difficulty_level);
CREATE INDEX idx_category ON exercise_solutions(category);
CREATE INDEX idx_created_at ON exercise_solutions(created_at);
