-- 练习草稿表
CREATE TABLE exercise_solution_drafts (
id SERIAL PRIMARY KEY,
exercise_solution_id INTEGER NOT NULL,
draft_content TEXT,
created_by VARCHAR(100),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
is_deleted BOOLEAN DEFAULT FALSE,

    -- 外键约束，关联练习解答表
    CONSTRAINT fk_exercise_solution_draft FOREIGN KEY (exercise_solution_id) 
        REFERENCES exercise_solutions(id) ON DELETE CASCADE,
    
    -- 唯一索引：一个练习解答只能有一个草稿
    CONSTRAINT uk_exercise_solution_id UNIQUE (exercise_solution_id)
);

-- 创建索引
CREATE INDEX idx_exercise_solution_drafts_exercise_id ON exercise_solution_drafts(exercise_solution_id);
CREATE INDEX idx_exercise_solution_drafts_is_deleted ON exercise_solution_drafts(is_deleted);
